- (Plugin?) method_missing(method_sym, *args) { ... }

Catches calls to undefined methods and checks whether they are of the form <defined method>?. If so, then it will call the defined method in a begin-rescue block and catch any errors and send events tagged as :error instead of passing the exception on to the caller.

def method_missing(method_sym, *args, &block)
  if method_sym.to_s =~ /^(.*)\?$/
    begin
      send($1, *args, &block)
    rescue => e
      notify(:msg => e.to_s, :tags => :error)
    end
  else
    super
  end
end

def notify(params)
  if @environment
    @environment.notify(params)
  elsif params[:msg] && params[:tags]
    # Fallback to $stdout - this is mostly for messages during setup
    $stdout.puts '[' + arraify(params[:tags]).join(',') + ']' + params[:msg]
  end
end

def respond_to?(method_sym, include_private = false)
  if method_sym.to_s =~ /^(.*)\?$/
    super($1, include_private)
  else
    super
  end
end
