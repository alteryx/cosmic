def arrayify(obj)
  if obj
    if obj.is_a?(Array)
      obj
    else
      [obj]
    end
  else
    []
  end
end

def require_with_hint(what, hint)
  begin
    require what
  rescue LoadError => e
    puts hint
    raise e
  end
end

- (Object?) with(obj) { ... }

Runs a block in the context of an object. This is similar to the with keyword in Pascal or JavaScript. For example:

str = "Hello World"
with str do
  puts reverse
end

It also supports blocks with arity of 1:

str = "Hello World"
with str do |my_str|
  puts my_str.reverse
end

Parameters:

  • obj (Object, nil) —

    The object that serves as the context for the block. If nil then the block won't be executed

Yields:

  • A block of arity 0 or 1 that will be executed in the context of the given object

Returns:

  • (Object, nil) —

    The return value of the block if any. If the block is not executed because the object is nil, then nothing will be returned

def with(obj, &block)
  if obj && block_given?
    if block.arity == 1
      yield obj
    else
      obj.instance_eval(&block)
    end
  end
end

- (Object?) with_available(obj) { ... }

This is the same as #with except that it will only run the block in the context of the object if the object is not an instance of Cosmic::HoneyPot. In effect this means that you'd use this in cases where a plugin is not configured but used in a script and you don't want to run code using the plugin at all.

Parameters:

  • obj (Object, nil) —

    The object that serves as the context for the block. If nil or an instance of Cosmic::HoneyPot then the block won't be executed

Yields:

  • A block of arity 0 or 1 that will be executed in the context of the given object

Returns:

  • (Object, nil) —

    The return value of the block if any. If the block is not executed because the object is nil, then nothing will be returned

def with_available(obj, &block)
  if obj && !obj.is_a?(Cosmic::HoneyPot) && block_given?
    if block.arity == 1
      yield obj
    else
      obj.instance_eval(&block)
    end
  end
end
